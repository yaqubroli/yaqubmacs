** Save changes

Eval this elisp code to tangle everything to its proper place.
#+begin_src emacs-lisp
  (dolist (file (directory-files "." t "\\.org$"))
    (with-current-buffer (find-file-noselect file)
      (let ((tangle-file (org-entry-get nil "tangle")))
        (when tangle-file
          (org-babel-tangle)))))
  (load-file "~/.emacs.d/init.el")
#+end_src
